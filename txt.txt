if - otomatis
	if - date > now
		validation error

	else if - date = now
		if - last_num not exist
			last_num = 1

		else if - last_num exist
			last_num + 1

	else if - date < now
		if - last_num not exist
			last_num = 1 & sisipan = a

		else if - last_num exist
			if - (last_num + 1) not exist
				last_num + 1

			else if - sisipan not exist
				last_num & sisipan = a

			else if - sisipan exist
				last_num & (sisipan + 1)

	else if - date = now
